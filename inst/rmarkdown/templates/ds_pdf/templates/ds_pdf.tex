%-----------------------------------------------------------
% This report style consists of a *banner* at the top of the
% page followed by the report header block.
%
%
%
%Create the header with markdown variables.
%
% A table (with no borders) aligns the text well.
%------------------------------------------------------

% We need the titling package to make the title work.
\usepackage{titling}

% We overload the title definitions for the purposes of
% the report header. Note, the date is emptied out so
% that \maketitle only creates the title block (banner) and
% the author block (which has the report header).
\date{}
\title{}

% We want the title to start at the top of the page, so within the
% titling package you do this with \droptitle.
\setlength{\droptitle}{-1in}

% Set the banner in the title field.

\title{\includegraphics[width=$if(banner_width)$ $banner_width$ $else$ 6.5in $endif$]{$banner$}}
$endif$

% Author field is the report header table. The titling package provides
% a way to have pre/post formatting code. The pre is to setup the
% tabular environment for the report output.

 \begin{center} % just for vertical spacing and killing indent
 \begin{tabular*}{\textwidth}{ll}
}
